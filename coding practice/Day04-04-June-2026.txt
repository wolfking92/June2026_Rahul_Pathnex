Ansible - Install Multiple Packages

---
- name: Install Packages for Pathnex
hosts: all
become: yes

tasks: 
    - name: Install tools
      yum:
        name: 
            - git
            - wget
            - python 
        state: present

Jenkins Pipeline — Parallel Build & Test

pipeline {
    agent any
    stages {
        stage('Parallel Tasks') {
            parallel {
                stage('Build') {
                    steps {
                        sh 'mvn clean compile'
                    }
                }
                stage('Test') {
                    steps {
                        sh 'mvn test'
                    }
                }
            }
        }
    }
}




GitLab CI — Parallel Jobs

stages:
  - build-test

maven-build:
  stage: build-test
  image: maven:3.8.1-jdk-17
  script:
    - git clone https://github.com/Pathnex/sample-java-app.git
    - cd sample-java-app
    - mvn clean compile
  parallel: 1

maven-test:
  stage: build-test
  image: maven:3.8.1-jdk-17
  script:
    - cd sample-java-app
    - mvn test
  parallel: 1
